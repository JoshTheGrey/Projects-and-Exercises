# Ask the user for a string and print out whether this is a
# string is a palindrome or not. (A palindrome is a string
# that reads the same forwards and backwards.)

word = str(input("Enter a word and the computer will determine if the word is a Palindrome:\n"))
word2 = word[::-1]
print(word2)
if word[0:] == word2[0:]:
    print(True)
else:
    print(False)
